What is Nala?

Nala is an ORM that handles your GraphQL requests and provides you with queries and mutations for your schemas. We take care of the communication between the server and the database so that you can focus on developing clean and powerful applications.

Example of a Nala Server

var Nala = require('nala');
var graphQLHandler = Nala(Schema, 'postgres://localhost/test');
app.post('/graphql', graphQLHandler);

Example Nala Schema

var userType = new GraphQLObjectType({
 	name: 'user',
 	description: 'this is the user type',
 	fields: {
   	  'name': {type: GraphQLString},
   	  'age': {type: GraphQLInt}
 	}
});

Example Nala Component

var query = {
  'query' : `query
              queryUser($name:String) {
               getUser(name: $name) {
                  name
                  age
               }
             }`,
  'variables' : {'name': this.state.name}
}

//send it to graphql
$.post('/graphql', query, function(response){
 console.dir(response.data);
});
